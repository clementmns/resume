% \resumeEntry – org + location row, with leading gap (use for 2nd+ entries)
\newcommand{\resumeEntry}[2]{%
    \vspace{6pt}%
    \begin{tabular*}{\textwidth}[t]{l@{\extracolsep{\fill}}r}%
        \textbf{#1} & \textbf{\small #2}%
    \end{tabular*}%
    \vspace{-2pt}%
}

% \resumeFirstEntry – org + location row, no leading gap (use for 1st entry)
\newcommand{\resumeFirstEntry}[2]{%
    \begin{tabular*}{\textwidth}[t]{l@{\extracolsep{\fill}}r}%
        \textbf{#1} & \textbf{\small #2}%
    \end{tabular*}%
    \vspace{-2pt}%
}

% \resumeRole – role / degree + date range, optional grade: \resumeRole[GPA: X]{title}{dates}
\newcommand{\resumeRole}[3][]{%
    \begin{tabular*}{\textwidth}{l@{\extracolsep{\fill}}r}%
        \textbf{\textit{\small #2}}\ifstrempty{#1}{}{\small\textbf{\ --\  #1}} & \textbf{\textit{\small #3}}%
    \end{tabular*}%
    \vspace{1pt}%
}

% \innerSectionGap – gap between two roles at the same org, or two schools
\newcommand{\innerSectionGap}{\vspace{5pt}}

% \resumeRoleOnly – second role under the same org, optional grade: \resumeRoleOnly[GPA: X]{title}{dates}
\newcommand{\resumeRoleOnly}[3][]{%
    \innerSectionGap%
    \begin{tabular*}{\textwidth}{l@{\extracolsep{\fill}}r}%
        \textbf{\textit{\small #2}}\ifstrempty{#1}{}{\small\ --\ #1} & \textbf{\textit{\small #3}}%
    \end{tabular*}%
    \vspace{1pt}%
}

% \resumeItemText – plain description line (Education)
\newcommand{\resumeItemText}[1]{%
    {\small #1}\par%
}

% Bullet-item list for accomplishments under a role
\newcommand{\resumeItemListStart}{%
    \begin{itemize}[%
        label={\scriptsize$\bullet$},%
        leftmargin=1.5em,%
        itemsep=1pt,%
        topsep=2pt,%
        partopsep=0pt,%
        parsep=0pt%
    ]%
}
\newcommand{\resumeItemListEnd}{\end{itemize}}

\newcommand{\resumeItem}[1]{\item\small{#1}}

% Flat label-less list for Projects and Certifications
\newcommand{\resumeDescListStart}{%
    \begin{itemize}[%
        leftmargin=0pt,%
        label={},%
        itemsep=3pt,%
        topsep=0pt,%
        partopsep=0pt,%
        parsep=0pt%
    ]%
}
\newcommand{\resumeDescListEnd}{\end{itemize}}

\newcommand{\resumeItemWithDesc}[2]{%
    \item{\textbf{\small #1}{\small : #2}}%
}

\newcommand{\resumeDescItem}[1]{%
    \item{\small #1}%
}
