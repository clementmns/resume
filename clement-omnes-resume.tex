% \embd - bold text
\newcommand{\embd}[1]{{\fontseries{sb}\selectfont #1}}

% \emsb - semibold text
\newcommand{\emsb}[1]{{\fontseries{sb}\selectfont #1}}

% \emmd – medium weight text
\newcommand{\emmd}[1]{{\fontseries{m}\selectfont #1}}

% \iconWithText - Icon followed by text
\newcommand{\iconWithText}[3][0.5em]{%
    \footnotesize\raisebox{0.1ex}{\scalebox{0.8}{#2}}\hspace{#1}#3%
}

% \resumeEntry – org + location row, with leading gap
\newcommand{\resumeEntry}[2]{%
    \begin{tabular*}{\textwidth}[t]{l@{\extracolsep{\fill}}r}%
        \embd{#1} & \small #2%
    \end{tabular*}%
}

% \resumeRole – role / degree + date range, optional grade: \resumeRole[GPA: X]{title}{dates}
\newcommand{\resumeRole}[3][]{%
    \begin{tabular*}{\textwidth}[t]{l@{\extracolsep{\fill}}r}%
        \embd{\textit{\small #2}}\ifstrempty{#1}{}{\small\embd{\ --\  #1}} & \textit{\small #3}%
    \end{tabular*}%
}

% \resumeItemText – plain description line
\newcommand{\resumeItemText}[1]{%
    {\small #1}\par%
}

% Bullet-item list
